1対N, N対N

serializer.py
from rest_framework import serializers
from .models import Shop, Item

class ShopSerializer(serializers.ModelSerializer):
    class Meta:
        model = Shop
        fields = ["id", "name"]
        read_only_fields = ('id',)

class ItemSerializer(serializers.ModelSerializer):
    shop = ShopSerializer

    class Meta:
        model = Item
        fields = ["id", "shop", "shop_id", "name", "price", "stock"]
        read_only_fields = ('id',)

この例ではShop1に対してItemN

子要素とページネーション

serializer.py
from rest_framework import serializers
from django.core.paginator import Paginator
from .models import Shop, Item

class ShopSerializer(serializers.ModelSerializer):
    paginated_items = serializers.SerializerMethodField()

    class Meta:
        model = Shop

    def get_paginated_items(self, instance):
        if instance is None:
            return {
                'prev': False,
                'next': False,
                'data': []
            }
        items = Item.objects.filter(item=instance).order_by('-created_at')
        if items.count() == 0:
            return {
                'prev': False,
                'next': False,
                'data': []
            }
        paginator = Paginator(items, 1ページあたりの要素数)
        page = paginator.page(1)
        serializer = ItemSerializer(page, many=True)
        result = {
            'prev': page.has_previous(),
            'next': page.has_next(),
            'data': serializer.data
        }
    return result

class ItemSerializer(serializers.ModelSerializer):
    shop = ShopSerializer

    class Meta:
        model = Item
